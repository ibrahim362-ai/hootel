## Routing Setup (React Router v6 Example)

```jsx
import { BrowserRouter, Routes, Route } from "react-router-dom";

// Admin
import AdminDashboard from "./pages/admin/Dashboard";
import Rooms from "./pages/admin/Rooms";
import Restaurant from "./pages/admin/Restaurant";
import Gym from "./pages/admin/Gym";

// Employer - Rooms
import EmployerRoomsDashboard from "./pages/employer/rooms/Dashboard";
import Tasks from "./pages/employer/rooms/Tasks";
import Shifts from "./pages/employer/rooms/Shifts";

// Employer - Restaurant
import EmployerRestaurantDashboard from "./pages/employer/restaurant/Dashboard";
import Orders from "./pages/employer/restaurant/Orders";
import Inventory from "./pages/employer/restaurant/Inventory";

// Employer - Gym
import EmployerGymDashboard from "./pages/employer/gym/Dashboard";
import TrainerSchedule from "./pages/employer/gym/TrainerSchedule";
import Classes from "./pages/employer/gym/Classes";

// Public (User)
import Home from "./pages/user/Home";
import RoomsPage from "./pages/user/Rooms";
import RestaurantPage from "./pages/user/Restaurant";
import GymPage from "./pages/user/Gym";
import Rewards from "./pages/user/Rewards";
import Profile from "./pages/user/Profile";

export default function App() {
  return (
    <BrowserRouter>
      <Routes>
        {/* Admin */}
        <Route path="/admin/dashboard" element={<AdminDashboard />} />
        <Route path="/admin/rooms" element={<Rooms />} />
        <Route path="/admin/restaurant" element={<Restaurant />} />
        <Route path="/admin/gym" element={<Gym />} />

        {/* Employer - Rooms */}
        <Route path="/employer/rooms/dashboard" element={<EmployerRoomsDashboard />} />
        <Route path="/employer/rooms/tasks" element={<Tasks />} />
        <Route path="/employer/rooms/shifts" element={<Shifts />} />

        {/* Employer - Restaurant */}
        <Route path="/employer/restaurant/dashboard" element={<EmployerRestaurantDashboard />} />
        <Route path="/employer/restaurant/orders" element={<Orders />} />
        <Route path="/employer/restaurant/inventory" element={<Inventory />} />

        {/* Employer - Gym */}
        <Route path="/employer/gym/dashboard" element={<EmployerGymDashboard />} />
        <Route path="/employer/gym/trainer-schedule" element={<TrainerSchedule />} />
        <Route path="/employer/gym/classes" element={<Classes />} />

        {/* Public/User */}
        <Route path="/" element={<Home />} />
        <Route path="/rooms" element={<RoomsPage />} />
        <Route path="/restaurant" element={<RestaurantPage />} />
        <Route path="/gym" element={<GymPage />} />
        <Route path="/rewards" element={<Rewards />} />
        <Route path="/profile" element={<Profile />} />
      </Routes>
    </BrowserRouter>
  );
}
```
